    (defun get-make ()
  (or (getenv "MAKE") "make"))

(define-symbol-macro MAKE (get-make))

(defmethod extract-phony-targets ((path string))
  (extract-phony-targets (pathname path)))

(defmethod extract-phony-targets ((path pathname))
  (if (directory-pathname-p path)
      (extract-phony-targets
       (merge-pathnames* #.(make-pathname :name "Makefile") path))
      (~>> path
           makefile-print-database
           lines
           (filter (op (string^= ".PHONY" _)))
           (mappend (op (rest (tokens _))))
           nub)))

(defun makefile-print-database (makefile)
  (check-type
   makefile
   (and tame-pathname file-pathname absolute-pathname physical-pathname))
  (assure string
    (run-program-in-directory
     (pathname-directory-pathname makefile)

(defun phony? (target dir)
  (member target
          (extract-phony-targets dir)
          :test #'equal))

(defun build-makefile-target (dir name)
  (run-program-in-directory
   dir
   (fmt "~a ~a" MAKE name)
   :output *makefile-output*))

(defun make-target-up-to-date? (dir name)
  (multiple-value-bind (stdout stderr status)
      (run-program-in-directory dir
                                (fmt "~a -q ~a" MAKE name)
                                :ignore-error-status t)
    (declare (ignore stdout stderr))
    (zerop status)))
